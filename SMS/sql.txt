CREATE TABLE [dbo].[UserRole]
(
	[ID] INT IDENTITY,
	ROLENAME VARCHAR(50) NOT NULL,
	PRIMARY KEY (ID)
)


CREATE TABLE [dbo].[Users]
(
	[ID] INT IDENTITY,
	USERNAME VARCHAR(255) NOT NULL UNIQUE,
	PASS VARCHAR(MAX) NOT NULL,
	ROLEID INT,
	CONTACTNO INT,
	PRIMARY KEY (ID),
	FOREIGN KEY (ROLEID) REFERENCES UserRole(ID)
)
